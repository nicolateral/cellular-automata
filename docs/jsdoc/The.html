Example

Automata.DefaultCellStateProvider = {

    getColor(state) {
        return state === 1 ? 'rgba(0, 0, 0, 0.7)' : 'rgb(255, 255, 255)';
    },

    getNextState(cell) {
        var count = 0,
            state;

        // Count each neighbors of state === 1
        cell.getNeighbors().forEach(function(neighbor) {
            if (neighbor.getState() === 1) {
                count += 1;
            }
        });

        // Here is the rule
        if (cell.getState() === 0) {
            if (count === 3) {
                return 1;
            }
        } else {
            if (!(count === 2 || count === 3)) {
                return 0;
            }
        }

        // Return
        return cell.getState();
    }
};
