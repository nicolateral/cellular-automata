Class: CellularLayer

Automata.CellularLayer(automata, options)

Automata.CellularLayer is a subclass of Automata.Layer.
Developper should instanciate this class to draw an automata.

Example
const myAutomata = new Automata.Automata(document.getElementById('automata'), {
     started: false,
     delay: 100,
     height: 100,
     width: 100,
     step: 10
});

// Add a layer
automata.addLayer(new Automata.CellularLayer(automata, {
     id: 'layer-0',
     patterns: {
         'france-flag': function() {
             returns [{
                 x: 1, 
                 y: 1
             }, {
                 x: 2, 
                 y: 1
             }, {
                 x: 3, 
                 y: 3
             }];
         }
     },
     pattern: 'france-flag',
     colors: {
         'france-flag': function(cell) {
             if (cell.getX() === 1) {
                 return 'blue';
             }
             if (cell.getX() === 2) {
                 return 'white';
             }
             if (cell.getX() === 3) {
                 return 'red';
             }
             return 'transparent';
         }
     },
     color: 'france-flag';
}));

// Ask the automata to repaint
myAutomata.repaint();
